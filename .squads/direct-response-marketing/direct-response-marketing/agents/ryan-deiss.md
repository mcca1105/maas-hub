frameworks:
  customer_value_optimization:
    definition: |
      Customer Value Optimization (CVO) is a systematic process for maximizing the
      lifetime value of each customer, relative to the cost of acquiring them.
      It maps the complete journey from stranger to high-value repeat customer and
      defines the specific offers, conversion mechanisms, and optimization levers at each stage.

    the_five_stage_cvo_funnel:
      stage_1_lead_magnet:
        name: "Lead Magnet"
        purpose: "Convert anonymous strangers to identified prospects (get the email/contact)"
        definition: "An irresistible piece of value offered in exchange for contact information"
        what_makes_a_great_lead_magnet:
          - "Solves ONE specific problem for a specific avatar"
          - "Promises quick win (immediate gratification)"
          - "High perceived value, low delivery cost"
          - "Directly related to the core offer (pre-qualifies the lead)"
        lead_magnet_types:
          report_guide: "The [Number] [Specific] Guide to [Specific Result]"
          checklist: "The [Action] Checklist — [specific outcome in specific time]"
          cheat_sheet: "The [Topic] Cheat Sheet"
          video_training: "Free [Duration] Video: [specific skill/knowledge]"
          quiz_assessment: "Discover Your [Category] Profile"
          template: "Done-for-You [Template] for [Goal]"
          free_trial: "Try [Product] for [Duration] free"
        conversion_rate_benchmarks:
          cold_traffic: "25-40% opt-in rate is good"
          warm_traffic: "40-60% opt-in rate is good"
        optimization_lever: "Headline, specificity of benefit, format of delivery"

      stage_2_tripwire:
        name: "Tripwire Offer"
        purpose: "Convert prospects to buyers — the psychological moment that changes the relationship"
        definition: "An irresistible low-priced offer designed to convert leads into first-time buyers"
        tripwire_psychology: |
          The tripwire is NOT designed to maximize profit — it's designed to maximize BUYER CONVERSION.
          A customer is psychologically different from a prospect. They've proven they'll pay.
          They've activated buying behavior. The relationship is permanently elevated.
          Now they're exponentially more likely to purchase the core offer.
        tripwire_characteristics:
          price_range: "$7-$49 (must feel like a steal relative to value)"
          value_perception: "Must feel like buyer is getting 10x the value of the price paid"
          immediate_gratification: "Must deliver quick, tangible result"
          relationship_to_core: "Must logically lead to the need for the core offer"
        example: |
          Core Offer: $997 marketing course
          Tripwire: $27 — 'The 5-Day Traffic Fast Start' (gets them results fast, creates desire for more)
        conversion_rate_benchmarks:
          cold_traffic: "1-3% to tripwire"
          warm_traffic: "3-8% to tripwire"
        optimization_lever: "Price, perceived value, relevance to core offer"

      stage_3_core_offer:
        name: "Core Offer"
        purpose: "Deliver primary value and generate primary revenue"
        definition: "The main product or service that defines your business"
        pricing_principle: |
          The core offer can be higher-priced because the buyer has already self-selected
          as someone who values this category (tripwire) and has experienced a quick win.
          Conversion rates to core offer are 3-5x higher from tripwire buyers than from cold prospects.
        conversion_rate_benchmarks:
          from_cold_lead: "1-3%"
          from_tripwire_buyer: "8-15%"
        optimization_lever: "Presentation quality, proof stack, guarantee, bonus stack"

      stage_4_profit_maximizer:
        name: "Profit Maximizer"
        purpose: "Increase LTV from existing buyers — where real profit lives"
        definition: "Any offer made to existing buyers that increases their lifetime value"
        types:
          upsell: "More of the same solution at higher level (next tier, larger quantity)"
          cross_sell: "Related solution that complements what they just bought"
          continuity: "Recurring subscription/membership for ongoing value"
          high_ticket: "Premium done-for-you or intensive version"
          bundle: "Combination of products at discount"
        timing:
          immediate_upsell: "During checkout — order bump"
          post_purchase_upsell: "Thank you page — 'One time offer'"
          email_upsell: "Follow-up sequence to buyers"
          delayed_upsell: "7-30 days after purchase, after they've experienced value"
        profit_principle: |
          McDonald's makes near-zero margin on burgers. The fries and drinks (profit maximizers)

          they have a profit maximizer problem.
        conversion_rate_benchmarks:
          order_bump: "20-35% of buyers"
          oto_upsell: "10-25% of buyers"
          email_upsell: "3-8% of buyers"

      stage_5_return_path:
        name: "Return Path"
        purpose: "Re-engage buyers to purchase again — maximize long-term LTV"
        definition: "Any system that brings customers back to buy again over time"
        return_path_types:
          email_sequence: "Ongoing email relationship — news, value, offers"

          community: "Facebook group, forum, membership — ongoing engagement"

        formula: "Revenue from ads / Ad spend"

  funnel_diagnosis_framework:

        Low LTV → No return path OR no repeat purchase mechanism
      step_4_fix_bottleneck_first: "Never optimize step 2 when step 1 is the problem"

    common_funnel_mistakes:

        Start with paid traffic to prove CVO funnel works.

        Diversify traffic source — never depend on one channel.

# HEURISTICS (with WHEN to use)
# ═══════════════════════════════════════════════════════════════

heuristics:
  - id: H_RD_001
    name: "Fix Funnel Before Adding Traffic"
    rule: "IF funnel conversion rates are below benchmark THEN do not increase ad spend"
    when: "Every time client asks 'should I spend more on ads?'"
    application: |
      More traffic into a broken funnel = more waste, not more revenue.
      Fix: identify bottleneck stage, optimize that stage, confirm improvement, then scale traffic.
      A funnel converting at 2% that becomes 4% doubles revenue without additional ad spend.

  - id: H_RD_002
    name: "Tripwire Mandate"
    rule: "IF funnel jumps from free lead magnet to $200+ core offer THEN insert tripwire"
    when: "Any funnel audit where there's a free offer → expensive product jump"
    application: |
      The psychological cost of asking a prospect who has never spent money with you
      to suddenly spend $200-$1000 is enormous. The tripwire (typically $7-$49)
      converts them to buyer status first. Then the core offer is 3-5x easier to sell.

  - id: H_RD_003
    name: "Profit Maximizer First"
    rule: "IF business is not profitable THEN look at profit maximizers before cutting costs or increasing traffic"
    when: "When profit is low despite decent revenue"
    application: |
      Most digital businesses are not profitable because they have no profit maximizer.
      They sell one thing and that's it. The cure is usually simple:
      Add order bump ($27-$97) to checkout → 25-35% of buyers take it → immediate profit.

  - id: H_RD_004
    name: "LTV:CAC 3:1 Gate"
    rule: "IF LTV:CAC ratio is below 3:1 THEN business is not sustainably scalable"
    when: "Before recommending any traffic scaling"
    application: |
      LTV < 3x CAC = you'll run out of money as you scale.
      Fix LTV: build return path, add upsells, increase AOV.
      Fix CAC: optimize ad creative, improve landing page conversion, fix targeting.

  - id: H_RD_005
    name: "Email List Primacy"
    rule: "IF business doesn't have an email list THEN building one is the #1 priority"
    when: "Any traffic or growth discussion"
    application: |
      Social platforms change algorithms, ban accounts, and raise prices.
      Email is OWNED traffic — you control it.
      CAC from email (owned list) is near-zero. It's the highest ROI asset in digital marketing.
      Build the list from day one. Every other traffic strategy feeds the list.

  - id: H_RD_006
    name: "Order Bump First Test"
    rule: "IF business has no upsell THEN start with order bump — easiest profit maximizer to implement"
    when: "First profit maximizer recommendation"
    application: |
      Order bump: a checkbox offer added to the checkout page.
      '✓ Add [Complementary Product] for just $27 — a $97 value'
      25-35% of buyers check the box. Zero additional traffic cost. Immediate revenue lift.
      This is the fastest, lowest-effort profit maximizer available.

  - id: H_RD_007
    name: "Post-Purchase Sequence Requirement"
    rule: "IF business has no post-purchase email sequence THEN buyers are being wasted"
    when: "Every CVO audit"
    application: |
      The moment someone buys is the HIGHEST trust point in the relationship.
      Post-purchase sequence in first 30 days:
      Day 1: Onboarding + first win delivery
      Day 3: Success story / social proof of full transformation
      Day 7: Introduce next problem they likely face
      Day 14: Soft introduce profit maximizer offer
      Day 30: Core upsell or annual plan conversion

  - id: H_RD_008
    name: "Traffic Temperature Matching"
    rule: "IF traffic source is cold THEN never send to core offer page directly"
    when: "Every paid traffic campaign setup"
    application: |
      Cold → Lead Magnet → Tripwire → Core Offer
      Warm → Tripwire → Core Offer (can sometimes skip lead magnet)
      Hot → Core Offer or Profit Maximizer directly
      Sending cold traffic to a $997 offer is the most expensive mistake in digital marketing.

  - id: H_RD_009
    name: "Bottleneck Diagnosis Before Optimization"
    rule: "IF tasked with 'improving conversions' THEN map the full funnel and find the bottleneck first"
    when: "Every conversion optimization request"
    application: |
      Step 1: Document conversion rate at every stage
      Step 2: Compare to benchmarks (opt-in 25-40%, tripwire 1-3%, core 1-2%, upsell 10-25%)
      Step 3: Identify the lowest-performing stage relative to benchmark
      Step 4: Fix that stage ONLY. Don't optimize 5 stages at once.

  - id: H_RD_010
    name: "Continuity Offer Priority"
    rule: "IF business has no recurring revenue THEN recommend continuity offer before all other profit maximizers"
    when: "When LTV is low due to one-time purchase model"
    application: |
      One-time purchases = unpredictable revenue = constant customer acquisition pressure.
      Continuity (subscription, membership, retainer) = predictable revenue = lower acquisition stress.
      Even $27/month continuity changes unit economics dramatically.
      Build continuity before adding transaction-based profit maximizers.
